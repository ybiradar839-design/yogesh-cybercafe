// localStorage key
const STORAGE_KEY = 'yogesh_custom_portals_v1';

function loadCustomPortals(){
  try{
    const raw = localStorage.getItem(STORAGE_KEY);
    if(!raw) return [];
    return JSON.parse(raw);
  }catch(e){return []}
}
function saveCustomPortals(list){
  localStorage.setItem(STORAGE_KEY, JSON.stringify(list));
}

function getAllPortals(){
  const customs = loadCustomPortals();
  // show custom portals first
  return [...customs, ...basePortals];
}

function makeCard(p, isCustom=false){
  const div = document.createElement('div');
  div.className='card';
  div.dataset.title = p.title.toLowerCase();
  div.dataset.tags = (p.tags||[]).join(' ').toLowerCase();
  const title = `<h3>${p.title}${isCustom?" <small style='font-size:0.8rem;color:var(--muted)'> (Custom)</small>":""}</h3>`;
  const tags = `<div class="tags">${(p.tags||[]).join(' Â· ')}</div>`;
  const deleteBtn = isCustom?`<button class="link" onclick="deleteCustom('${encodeURIComponent(p.url)}')">Delete</button>`:'';
  const actions = `<div class="actions"><button class="link primary" onclick="openSafe('${encodeURIComponent(p.url)}')">Open</button><button class="link" onclick="copyLink('${p.url}')">Copy URL</button>${deleteBtn}</div>`;
  div.innerHTML = title + tags + actions;
  return div;
}

function render(){
  const list = document.getElementById('list');
  list.innerHTML='';
  const customs = loadCustomPortals();
  // custom first
  customs.forEach(p=> list.appendChild(makeCard(p,true)));
  basePortals.forEach(p=> list.appendChild(makeCard(p,false)));
}

function filter(){
  const q = document.getElementById('q').value.trim().toLowerCase();
  const cards = document.querySelectorAll('#list .card');
  cards.forEach(c=>{
    const text = c.dataset.title + ' ' + c.dataset.tags;
    c.style.display = text.includes(q) ? '' : 'none';
  });
}
function resetFilters(){document.getElementById('q').value='';filter();}

function copyLink(url){navigator.clipboard?.writeText(url).then(()=>alert('URL copy ho gaya')).catch(()=>prompt('Copy manually, yeh URL hai:',url));}

function openSafe(encodedUrl){
  const url = decodeURIComponent(encodedUrl);
  try{
    const a = document.createElement('a');
    a.href = url;
    a.target = '_blank';
    a.rel = 'noopener noreferrer';
    a.style.display='none';
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    return;
  }catch(e){}
  try{const w = window.open(url, '_blank', 'noopener,noreferrer'); if(w) return;}catch(e){}
  try{window.location.href = url;}catch(err){
    alert('Browser ne external navigation block kar diya. Right-click -> Open link in new tab ya Copy URL karke manually paste karo.');
  }
}

// Add portal modal logic
function showAddModal(){
  const tpl = document.getElementById('modalTpl');
  const node = tpl.content.cloneNode(true);
  document.body.appendChild(node);
  const modal = document.getElementById('modal');
  const name = document.getElementById('portalName');
  const url = document.getElementById('portalURL');
  document.getElementById('cancelPortalBtn').onclick = () => modal.remove();
  document.getElementById('savePortalBtn').onclick = () => {
    const n = name.value.trim();
    let u = url.value.trim();
    if(!n || !u){alert('Dono fields bharna zaroori hai');return}
    if(!/^https?:\/\//i.test(u)) u = 'https://'+u;
    addCustom({title:n,url:u,tags:[]});
    modal.remove();
  };
}

function addCustom(p){
  const list = loadCustomPortals();
  // avoid dup by url
  if(list.find(x=>x.url===p.url)){
    alert('Ye portal pehle se added hai');
    render();
    return;
  }
  list.unshift(p);
  saveCustomPortals(list);
  render();
}

function deleteCustom(encodedUrl){
  const url = decodeURIComponent(encodedUrl);
  if(!confirm('Kya aap is custom portal ko delete karna chahte hain?')) return;
  let list = loadCustomPortals();
  list = list.filter(p=>p.url !== url);
  saveCustomPortals(list);
  render();
}
